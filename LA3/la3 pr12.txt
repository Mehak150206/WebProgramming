    <script>
        function validateSkills() {
            var skillsCBs = document.getElementsByName("skills");
            var selectedCount = 0;
            var selectedSkills = "";
            var message = "";

            for (let i = 0; i < skillsCBs.length; i++) {
                if (skillsCBs[i].checked) {
                    selectedCount++;

                    if (selectedSkills !== "") {
                        selectedSkills += ", ";
                    }
                    selectedSkills += skillsCBs[i].value;
                }
            }

            if (selectedCount < 3) {
                message = "<span class='error'>Please select at least 3 skills. You selected: " + selectedCount + "</span>";
            } else {
                message = "<span class='success'>✓ Valid! Selected skills: " + selectedSkills + "</span>";
            }

            document.getElementById("result").innerHTML = message;
        }
    </script>
