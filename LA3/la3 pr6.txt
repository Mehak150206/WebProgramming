<script>
    // Global array to store students
    var students = [];

    // Constructor function (Object)
    function Student(name, age, course) {
        this.name = name;
        this.age = age;
        this.course = course;

        // Object method
        this.display = function() {
            return "<td>" + this.name + "</td>" +
                   "<td>" + this.age + "</td>" +
                   "<td>" + this.course + "</td>";
        };
    }

    // Function to add student
    function addStudent(form) {
        // Get values from form
        var name = form.elements["name"].value;
        var age = form.elements["age"].value;
        var course = form.elements["course"].value;

        // Create object and store in array
        var studentObj = new Student(name, age, course);
        students.push(studentObj);

        // Update table dynamically
        displayStudents();

        // Reset form and focus
        form.reset();
        form.elements["name"].focus();
    }

    // Function to display students
    function displayStudents() {
        var table = document.getElementById("studentTable");

        // Reset table (keep header)
        table.innerHTML = "<tr><th>Name</th><th>Age</th><th>Course</th></tr>";

        // Loop through array
        for (var i = 0; i < students.length; i++) {
            var row = "<tr>" + students[i].display() + "</tr>";
            table.innerHTML += row;
        }
    }

    // Reset function
    function resetForm(form) {
        form.reset();
        form.elements["name"].focus();
    }
</script>
