    <script>
        function applyStyle(styleClass) {
            var message = document.getElementById("message");
            message.className = styleClass;

            document.getElementById("currentClass").innerHTML =
                "Current class: <code>" + styleClass + "</code>";
        }

        function combineStyles() {
            var message = document.getElementById("message");
            message.className = "italic bold small-cap";

            document.getElementById("currentClass").innerHTML =
                "Current classes: <code>italic bold small-cap</code>";
        }

        function clearStyles() {
            document.getElementById("message").className = "";
            document.getElementById("currentClass").innerHTML =
                "Current class: (none)";
        }

        function toggleHighlight() {
            var message = document.getElementById("message");
            var currentClasses = message.className;

            if (currentClasses.includes("highlight")) {
                message.className = currentClasses.replace(" highlight", "");
            } else {
                message.className = currentClasses + " highlight";
            }
        }
    </script>
