    <script>
        var topIndex = 2;

        function moveToTop(picture) {
            picture.style.zIndex = ++topIndex;
            updateZIndexDisplay();

            console.log("Moved to front: " + picture.className);
            console.log("New topIndex: " + topIndex);
        }

        function updateZIndexDisplay() {
            var tiger = document.querySelector(".tiger");
            var lemur = document.querySelector(".lemur");
            var panda = document.querySelector(".red-panda");

            var html = "<strong>Current Z-Index Values:</strong><br>";
            html += "Tiger: <span class='z-index-display'>" + tiger.style.zIndex + "</span> | ";
            html += "Lemur: <span class='z-index-display'>" + lemur.style.zIndex + "</span> | ";
            html += "Red Panda: <span class='z-index-display'>" + panda.style.zIndex + "</span>";

            document.getElementById("zIndexInfo").innerHTML = html;
        }
    </script>

        <strong>How it works:</strong><br>
        • <code>position: absolute</code> is required for z-index<br>
        • Global variable <code>topIndex</code> tracks highest z-index<br>
        • <code>++topIndex</code> (prefix increment) increases then assigns<br>
        • Higher z-index = closer to viewer (front)
