    <input type="checkbox" name="subject" value="Computer"> Computer<br><br>

    <button onclick="countSubjects()">Show Selection</button>

    <p id="result"></p>

    <script>
        function countSubjects() {
            var subjects = document.getElementsByName("subject");
            var selectedSubjects = []; // array
            var count = 0; // counter

            // Loop through checkboxes
            for (var i = 0; i < subjects.length; i++) {
                if (subjects[i].checked) {
                    selectedSubjects.push(subjects[i].value); // add to array
                    count++;
                }
            }

            // Display result
            if (count > 0) {
                document.getElementById("result").innerHTML =
                    "Selected Subjects: " + selectedSubjects.join(", ") +
                    "<br>Total Selected: " + count;
            } else {
                document.getElementById("result").innerHTML =
                    "No subjects selected";
            }
        }
    </script>
