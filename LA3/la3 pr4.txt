    <button onclick="calculateArea()">Calculate Area</button>

    <p id="result"></p>

    <script>
        function calculateArea() {
            // Get input values
            var lengthValue = document.getElementById("lengthValue").value;
            var breadthValue = document.getElementById("breadthValue").value;

            // Calculate area (loose typing handles conversion)
            var areaResult = lengthValue * breadthValue;

            // Display result
            document.getElementById("result").innerHTML =
                "Area of Rectangle: " + areaResult;
        }
    </script>
