    Enter Number: <input type="text" id="num"><br><br>

    <button onclick="printTable()">Generate Table</button>

    <p id="result"></p>

    <script>
        function printTable() {
            // Get input
            var n = document.getElementById("num").value;

            var i = 1;
            var output = "";

            // Do-while loop
            do {
                output += n + " x " + i + " = " + (n * i) + "<br>";
                i++;
            } while (i <= 10);

            // Display result
            document.getElementById("result").innerHTML = output;
        }
    </script>
