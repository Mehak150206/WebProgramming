    <script>
        function calculateTotal() {
            var checkedItems = document.querySelectorAll('input[name="product"]:checked');

            var total = 0;
            var itemList = "";

            for (let item of checkedItems) {
                var price = parseFloat(item.getAttribute("data-price"));
                total += price;

                if (itemList !== "") {
                    itemList += "<br>";
                }
                itemList += "• " + item.value + " - $" + price.toFixed(2);
            }

            var resultHTML = "<strong>Selected Items:</strong><br>" + itemList;
            resultHTML += "<div class='total'>Total: $" + total.toFixed(2) + "</div>";

            document.getElementById("orderSummary").innerHTML = resultHTML;
        }
    </script>
